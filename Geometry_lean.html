A M B ∧ Geo.Congruent A M M B def IsMedianIntersection (B E C F G : Geo.Point) : Prop := Geo.Collinear B G E ∧ Geo.Collinear C G F def IsSegmentIntersection (A P B C D : Geo.Point) : Prop := Geo.Collinear A D P ∧ Geo.Collinear B D C /-! ## Axioms & Properties -/ axiom SAS (V X Y V' X' Y' : Geo.Point) : Geo.Congruent V X V' X' → Geo.AngleCongruent X V Y X' V' Y' → Geo.Congruent V Y V' Y' → Geo.Congruent A B D C axiom congruent_transitivity (A D C B F : Geo.Point) : Geo.Congruent A D D C → Geo.Congruent C D B F → Geo.Congruent A D B F axiom CollinearSymmetry (A B C : Geo.Point) : Geo.Collinear A B C → Geo.Collinear C B A axiom CollinearRotate (A B C : Geo.Point) : Geo.Collinear A B C → Geo.Collinear A C B axiom collinear_trans (A G D P : Geo.Point) : Geo.Collinear A G P → Geo.Congruent B A D C /-! ## Derived Lemmas -/ theorem ParallelSwapFirstLine' (E G C P : Geo.Point) (h : Geo.Parallel E G C P) : Geo.Parallel G E C P := ParallelSwapFirstLine E G C P h theorem MedianIntersectionCollinearCGF (B E C F G : Geo.Point) (h : IsMedianIntersection B E C F G) : Geo.Collinear C G F := h.right theorem MedianIntersectionCollinearBGE (B E C F G : Geo.Point) (h : IsMedianIntersection B E C F G) : Geo.Collinear B G E := h.left theorem SegmentIntersectionCollinearADP (A P B C D : Geo.Point) (h : IsSegmentIntersection A P B C D) : Geo.Collinear A D P := h.left theorem SegmentIntersectionCollinearBDC (A P B C D : Geo.Point) (h : IsSegmentIntersection A P B C D) : Geo.Collinear B D C := h.right theorem MidpointSymmetry (M A B : Geo.Point) (h : IsMidpoint M A B) : IsMidpoint M B A := by rcases h with ⟨hCol, hCong⟩ constructor · exact CollinearSymmetry A M B hCol · have h1 := CongruentReverseBoth A M M B hCong exact CongruentSymmetry M A B M h1 theorem MidsegmentParallel (V₁ V₂ V₃ M₁ M₂ : Geo.Point) (hM₁ : IsMidpoint M₁ V₁ V₃) (hM₂ : IsMidpoint M₂ V₂ V₃) : Geo.Parallel M₁ M₂ V₁ V₂ := by rcases ExtendSegment M₁ M₂ with ⟨T, hM₁M₂T, hSeg⟩ have hV₁M₁V₃ := hM₁.left have hV₃M₂V₂ := CollinearSymmetry V₂ M₂ V₃ hM₂.left have hVert := VerticalAngles V₃ M₂ M₁ V₂ M₂ T hV₃M₂V₂ hM₁M₂T have hVert' := AngleCongruentReverse V₃ M₂ M₁ V₂ M₂ T hVert have hSideM₁M₂M₂T := CongruentReverseFirst M₁ M₂ M₂ T hSeg have hSideV₂M₂M₂V₃ := CongruentReverseBoth V₂ M₂ M₂ V₃ hM₂.right have hSideV₃M₂M₂V₂ := CongruentReverseFirst V₃ M₂ M₂ V₂ (CongruentSymmetry M₂ V₂ V₃ M₂ hSideV₂M₂M₂V₃) have hSAS := SAS M₂ M₁ V₃ M₂ T V₂ hSideM₁M₂M₂T hVert' hSideV₃M₂M₂V₂ have hParV₁M₁V₂T := parallel_from_equal_angles V₁ V₃ M₁ V₂ M₂ T (CollinearRotate V₁ M₁ V₃ hV₁M₁V₃) hSAS.angleC have hSideM₁V₃TV₂ := hSAS.sideBC have hSideV₃M₁TV₂ : Geo.Congruent V₃ M₁ T V₂ := CongruentReverseFirst M₁ V₃ T V₂ hSideM₁V₃TV₂ have hSideV₃M₁V₂T : Geo.Congruent V₃ M₁ V₂ T := CongruentSwapSecond V₃ M₁ T V₂ hSideV₃M₁TV₂ have hSideV₁M₁V₂T := congruent_transitivity V₁ M₁ V₃ V₂ T hM₁.right hSideV₃M₁V₂T have hParallelogram := parallelogram_from_parallel_and_congruent V₁ V₂ T M₁ hParV₁M₁V₂T hSideV₁M₁V₂T have hParTM₁V₁V₂ := parallelogram_opposite_sides V₁ V₂ T M₁ hParallelogram exact collinear_parallel_trans M₁ M₂ T V₁ V₂ hM₁M₂T hParTM₁V₁V₂ /-! ## Main Theorem -/ theorem FinlayProof (A B C E F G P D : Geo.Point) (hE : IsMidpoint E A C) (hF : IsMidpoint F A B) (hG : IsMedianIntersection B E C F G) (hP : IsMidpoint G A P) (hD : IsSegmentIntersection A P B C D) : Geo.Congruent B D D C := by ------------------------------------------------------------------ -- Step 1. -- Apply the Midsegment Theorem ------------------------------------------------------------------ -- In triangle ABP, we obtain FG ∥ BP, -- and since C,G,F are collinear, CG ∥ BP also holds. have hFB := MidpointSymmetry F A B hF have hGP := MidpointSymmetry G A P hP have hFGparallelBP := MidsegmentParallel B P A F G hFB hGP have hCGparallelBP := collinear_parallel_substitute F G C B P (MedianIntersectionCollinearCGF B E C F G hG) hFGparallelBP -- In triangle ACP, we obtain GE ∥ PC, -- and since B,G,E are collinear, BG ∥ PC also holds. have hEA := MidpointSymmetry E A C hE have hGA := MidpointSymmetry G A P hP have hEGparallelCP := MidsegmentParallel C P A E G hEA hGA have hGEparallelCP := ParallelSwapFirstLine E G C P hEGparallelCP have hBGparallelCP := collinear_parallel_substitute2 G E B C P (MedianIntersectionCollinearBGE B E C F G hG) hGEparallelCP ------------------------------------------------------------------ -- Step 2. -- From two pairs of parallel opposite sides, -- we conclude that BPCG is a parallelogram. ------------------------------------------------------------------ have hParallelogram := ParallelogramFromFinlayParallel B P C G hCGparallelBP hBGparallelCP -- We have proven that BPCG is a parallelogram. ------------------------------------------------------------------ -- Step 3. -- Determine the intersection point of the parallelogram diagonals. ------------------------------------------------------------------ have hGDP := collinear_trans A G D P hP.left (SegmentIntersectionCollinearADP A P B C D hD) ------------------------------------------------------------------ -- Step 4. -- Property of parallelogram diagonals: -- diagonals bisect each other. ------------------------------------------------------------------ have hBD_eq_DC := parallelogram_diagonals_bisect B P C G D hParallelogram (SegmentIntersectionCollinearBDC A P B C D hD) hGDP -- We have proven that BD ≅ DC. ------------------------------------------------------------------ -- Step 5. -- We obtain BD = DC, which completes the proof. ------------------------------------------------------------------ exact hBD_eq_DC end Geometry
