structure Geometry where
  Point : Type
  Collinear : Point → Point → Point → Prop
  Congruent : Point → Point → Point → Point → Prop
  Parallel : Point → Point → Point → Point → Prop
  AngleCongruent : Point → Point → Point → Point → Point → Point → Prop

variable {Geo : Geometry}

/-! ### Basic Structures & Definitions -/

structure SASResult (A B C D E F : Geo.Point) where
  sideAB : Geo.Congruent A B D E
  sideBC : Geo.Congruent B C E F
  sideAC : Geo.Congruent A C D F
  angleA : Geo.AngleCongruent B A C E D F
  angleB : Geo.AngleCongruent A B C D E F
  angleC : Geo.AngleCongruent A C B D F E

def IsMidpoint (M A B : Geo.Point) : Prop :=
  Geo.Collinear A M B ∧ Geo.Congruent A M M B

def IsMedianIntersection (B E C F G : Geo.Point) : Prop :=
  Geo.Collinear B G E ∧ Geo.Collinear C G F

def IsSegmentIntersection (A P B C D : Geo.Point) : Prop :=
  Geo.Collinear A D P ∧ Geo.Collinear B D C

def OppositeSidesParallel (A B C D : Geo.Point) : Prop :=
  Geo.Parallel A B C D ∧ Geo.Parallel B C D A

def OppositeSidesCongruent (A B C D : Geo.Point) : Prop :=
  Geo.Congruent A B C D ∧ Geo.Congruent B C D A

def DiagonalsBisect (A B C D : Geo.Point) : Prop :=
  ∃ M : Geo.Point, IsMidpoint M A C ∧ IsMidpoint M B D

abbrev IsParallelogram (A B C D : Geo.Point) : Prop :=
  OppositeSidesParallel A B C D

def OnePairParallelCongruent (A B C D : Geo.Point) : Prop :=
  Geo.Parallel A D B C ∧ Geo.Congruent A D B C

def OppositeAnglesCongruent (A B C D : Geo.Point) : Prop :=
  Geo.AngleCongruent A B C C D A ∧ Geo.AngleCongruent B C D D A B

/-! ### Axioms & Properties -/

axiom SAS (V X Y V' X' Y' : Geo.Point) :
  Geo.Congruent V X V' X' → Geo.AngleCongruent X V Y X' V' Y' →
  Geo.Congruent V Y V' Y' → SASResult V X Y V' X' Y'

axiom CongruentSymmetry (A B C D : Geo.Point) : Geo.Congruent A B C D → Geo.Congruent C D A B
axiom CongruentSwapSecond (A B C D : Geo.Point) : Geo.Congruent A B C D → Geo.Congruent A B D C
axiom congruent_transitivity (A D C B F : Geo.Point) : Geo.Congruent A D D C → Geo.Congruent C D B F → Geo.Congruent A D B F
axiom CollinearSymmetry (A B C : Geo.Point) : Geo.Collinear A B C → Geo.Collinear C B A
axiom CollinearRotate (A B C : Geo.Point) : Geo.Collinear A B C → Geo.Collinear A C B
axiom collinear_trans (A G D P : Geo.Point) : Geo.Collinear A G P → Geo.Collinear A D P → Geo.Collinear G D P
axiom parallel_from_equal_angles (A C D B E F : Geo.Point) : Geo.Collinear A C D → Geo.AngleCongruent E C D E B F → Geo.Parallel A D B F
axiom collinear_parallel_substitute (F G C B P : Geo.Point) : Geo.Collinear C G F → Geo.Parallel F G B P → Geo.Parallel C G B P
axiom collinear_parallel_substitute2 (G E B C P : Geo.Point) : Geo.Collinear B G E → Geo.Parallel G E C P → Geo.Parallel B G C P
axiom OnePairParallelCongruentCriterion (A B F D : Geo.Point) : Geo.Parallel A D B F → Geo.Congruent A D B F → IsParallelogram A B F D
axiom ParallelogramOppositeSidesParallel (A B F D : Geo.Point) : IsParallelogram A B F D → Geo.Parallel D F A B
axiom ParallelogramDiagonalIntersectionMidpoint (B P C G D : Geo.Point) : IsParallelogram B P C G → Geo.Collinear B D C → Geo.Collinear G D P → IsMidpoint D B C
axiom ParallelogramDiagonalIntersectionMidpoints (B P C G D : Geo.Point) : IsParallelogram B P C G → Geo.Collinear B D C → Geo.Collinear G D P → IsMidpoint D B C ∧ IsMidpoint D G P
axiom ParallelogramDiagonalsBisectCore (A B C D : Geo.Point) : IsParallelogram A B C D → DiagonalsBisect A B C D
axiom ExtendSegment (D E : Geo.Point) : ∃ F : Geo.Point, Geo.Collinear D E F ∧ Geo.Congruent D E E F
axiom VerticalAngles (C E D B E F : Geo.Point) : Geo.Collinear C E B → Geo.Collinear D E F → Geo.AngleCongruent C E D B E F
axiom collinear_parallel_trans (D E F A B : Geo.Point) : Geo.Collinear D E F → Geo.Parallel D F A B → Geo.Parallel D E A B
axiom CongruentReverseFirst (A B C D : Geo.Point) : Geo.Congruent A B C D → Geo.Congruent B A C D
axiom AngleCongruentReverse (A B C D E F : Geo.Point) : Geo.AngleCongruent A B C D E F → Geo.AngleCongruent C B A F E D
axiom CongruentReverseBoth (A B C D : Geo.Point) : Geo.Congruent A B C D → Geo.Congruent B A D C
axiom ParallelSwapFirstLine (A B C D : Geo.Point) : Geo.Parallel A B C D → Geo.Parallel B A C D
axiom ParallelSwapSecondLine (A B C D : Geo.Point) : Geo.Parallel A B C D → Geo.Parallel A B D C
axiom ParallelSwapBoth (A B C D : Geo.Point) : Geo.Parallel A B C D → Geo.Parallel B A D C
axiom ParallelSymmetry (A B C D : Geo.Point) : Geo.Parallel A B C D → Geo.Parallel C D A B
axiom ParallelogramOppositeSidesCongruentCore (A B C D : Geo.Point) : IsParallelogram A B C D → OppositeSidesCongruent A B C D
axiom MidpointUnique (A B M N : Geo.Point) : IsMidpoint M A B → IsMidpoint N A B → M = N

theorem MedianIntersectionCollinearCGF (B E C F G : Geo.Point) (h : IsMedianIntersection B E C F G) : Geo.Collinear C G F := h.right
theorem MedianIntersectionCollinearBGE (B E C F G : Geo.Point) (h : IsMedianIntersection B E C F G) : Geo.Collinear B G E := h.left
theorem SegmentIntersectionCollinearADP (A P B C D : Geo.Point) (h : IsSegmentIntersection A P B C D) : Geo.Collinear A D P := h.left

theorem ParallelogramOppositeSidesCongruent (A B C D : Geo.Point) : IsParallelogram A B C D → OppositeSidesCongruent A B C D := by intro hPar; exact ParallelogramOppositeSidesCongruentCore A B C D hPar
theorem ParallelogramDiagonalsBisect (A B C D : Geo.Point) : IsParallelogram A B C D → DiagonalsBisect A B C D := by intro hPar; exact ParallelogramDiagonalsBisectCore A B C D hPar
theorem OnePairParallelCongruentRecognition (A B C D : Geo.Point) : OnePairParallelCongruent A B C D → IsParallelogram A B C D := by intro h; rcases h with ⟨hPar, hCong⟩; exact OnePairParallelCongruentCriterion A B C D hPar hCong

theorem MidpointSymmetry (M A B : Geo.Point) (h : IsMidpoint M A B) : IsMidpoint M B A := by
  rcases h with ⟨hCol, hCong⟩
  constructor
  · exact CollinearSymmetry A M B hCol
  · have h1 := CongruentReverseBoth A M M B hCong
    exact CongruentSymmetry M A B M h1

theorem MidsegmentParallel (V₁ V₂ V₃ M₁ M₂ : Geo.Point) (hM₁ : IsMidpoint M₁ V₁ V₃) (hM₂ : IsMidpoint M₂ V₂ V₃) : Geo.Parallel M₁ M₂ V₁ V₂ := by
  rcases ExtendSegment M₁ M₂ with ⟨T, hM₁M₂T, hSeg⟩
  have hV₁M₁V₃ := hM₁.left
  have hV₃M₂V₂ := CollinearSymmetry V₂ M₂ V₃ hM₂.left
  have hVert := VerticalAngles V₃ M₂ M₁ V₂ M₂ T hV₃M₂V₂ hM₁M₂T
  have hVert' := AngleCongruentReverse V₃ M₂ M₁ V₂ M₂ T hVert
  have hSideM₁M₂M₂T := CongruentReverseFirst M₁ M₂ M₂ T hSeg
  have hSideV₂M₂M₂V₃ := CongruentReverseBoth V₂ M₂ M₂ V₃ hM₂.right
  have hSideV₃M₂M₂V₂ := CongruentReverseFirst V₃ M₂ M₂ V₂ (CongruentSymmetry M₂ V₂ V₃ M₂ hSideV₂M₂M₂V₃)
  have hSAS := SAS M₂ M₁ V₃ M₂ T V₂ hSideM₁M₂M₂T hVert' hSideV₃M₂M₂V₂
  have hParV₁M₁V₂T := parallel_from_equal_angles V₁ V₃ M₁ V₂ M₂ T (CollinearRotate V₁ M₁ V₃ hV₁M₁V₃) hSAS.angleC
  have hSideM₁V₃TV₂ := hSAS.sideBC
  have hSideV₃M₁TV₂ : Geo.Congruent V₃ M₁ T V₂ := CongruentReverseFirst M₁ V₃ T V₂ hSideM₁V₃TV₂
  have hSideV₃M₁V₂T : Geo.Congruent V₃ M₁ V₂ T := CongruentSwapSecond V₃ M₁ T V₂ hSideV₃M₁TV₂
  have hSideV₁M₁V₂T := congruent_transitivity V₁ M₁ V₃ V₂ T hM₁.right hSideV₃M₁V₂T
  have hParallelogram := OnePairParallelCongruentCriterion V₁ V₂ T M₁ hParV₁M₁V₂T hSideV₁M₁V₂T
  have hParTM₁V₁V₂ := ParallelogramOppositeSidesParallel V₁ V₂ T M₁ hParallelogram
  exact collinear_parallel_trans M₁ M₂ T V₁ V₂ hM₁M₂T hParTM₁V₁V₂

theorem ParallelogramParallelRecognition (B P C G : Geo.Point) (hCGparallelBP : Geo.Parallel C G B P) (hBGparallelCP : Geo.Parallel B G C P) : IsParallelogram B P C G := by
  dsimp [IsParallelogram]
  have hBPparallelCG := ParallelSymmetry C G B P hCGparallelBP
  have hCPparallelBG := ParallelSymmetry B G C P hBGparallelCP
  constructor
  · exact hBPparallelCG
  · exact ParallelSwapBoth C P B G hCPparallelBG

theorem ParallelogramDiagonalsMidpoint (B P C G D : Geo.Point) (hPar : IsParallelogram B P C G) (hBDC : Geo.Collinear B D C) (hGDP : Geo.Collinear G D P) : IsMidpoint D B C := by
  have hMid := ParallelogramDiagonalIntersectionMidpoints B P C G D hPar hBDC hGDP
  exact hMid.left

/-! ### Main Theorem -/

theorem FinlayProof (A B C E F G P D : Geo.Point)
  (hE : IsMidpoint E A C) (hF : IsMidpoint F A B) (hG : IsMedianIntersection B E C F G)
  (hP : IsMidpoint G A P) (hD : IsSegmentIntersection A P B C D) : Geo.Congruent B D D C := by
  have hFB := MidpointSymmetry F A B hF
  have hGP := MidpointSymmetry G A P hP
  have hFGparallelBP := MidsegmentParallel B P A F G hFB hGP
  have hCGparallelBP := collinear_parallel_substitute F G C B P (MedianIntersectionCollinearCGF B E C F G hG) hFGparallelBP
  have hEA := MidpointSymmetry E A C hE
  have hGA := MidpointSymmetry G A P hP
  have hEGparallelCP := MidsegmentParallel C P A E G hEA hGA
  have hGEparallelCP := ParallelSwapFirstLine E G C P hEGparallelCP
  have hBGparallelCP := collinear_parallel_substitute2 G E B C P (MedianIntersectionCollinearBGE B E C F G hG) hGEparallelCP
  have hParallelogram := ParallelogramParallelRecognition B P C G hCGparallelBP hBGparallelCP
  have hGDP := collinear_trans A G D P hP.left (SegmentIntersectionCollinearADP A P B C D hD)
  have hMidBC := ParallelogramDiagonalsMidpoint B P C G D hParallelogram (SegmentIntersectionCollinearBDC A P B C D hD) hGDP
  exact hMidBC.right

end Geometry
