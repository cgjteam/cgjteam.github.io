namespace Geometry

universe u

structure Geo where
  Point : Type u
  Collinear : Point → Point → Point → Prop
  Parallel : Point → Point → Point → Point → Prop
  Congruent : Point → Point → Point → Point → Prop

variable (Geo : Geo)

axiom collinear_swap (A B C : Geo.Point) :
  Geo.Collinear A B C → Geo.Collinear A C B

def IsMidpoint (Geo : Geometry.Geo) (M A B : Geo.Point) : Prop :=
  Geo.Collinear A M B ∧ Geo.Congruent A M M B

theorem midpoint_collinear (Geo : Geometry.Geo) (A B M : Geo.Point) :
    IsMidpoint Geo M A B → Geo.Collinear A M B := by
  intro h
  exact h.left

def IsIntersection (A B C D P : Geo.Point) : Prop :=
  Geo.Collinear A P B ∧ Geo.Collinear C P D

def IsMedian (Geo : Geometry.Geo) (_ M B C : Geo.Point) : Prop :=
  IsMidpoint Geo M B C

/--
A median is currently represented only by the midpoint
of the opposite side. The first argument (vertex) is
kept for compatibility with the existing API.
-/
axiom MidsegmentTheorem (A B C M N : Geo.Point) :
  IsMidpoint Geo M A B →
  IsMidpoint Geo N A C →
  Geo.Parallel M N B C

def IsParallelogram (Geo : Geometry.Geo) (A B C D : Geo.Point) : Prop :=
  Geo.Parallel A B C D ∧ Geo.Parallel A D B C

axiom ParallelogramDiagonals (A B C D M : Geo.Point) :
  IsParallelogram Geo A B C D →
  Geo.Collinear A M C →
  Geo.Collinear B M D →
  IsMidpoint Geo M A C

theorem MidpointMedian (Geo : Geometry.Geo) (A B C M : Geo.Point) :
    IsMidpoint Geo M B C → IsMedian Geo A M B C := by
  intro h
  exact h

axiom ParallelCollinearLeft (A B C D E : Geo.Point) :
  Geo.Parallel A B D E →
  Geo.Collinear C A B →
  Geo.Parallel C B D E

axiom ParallelSymm (A B C D : Geo.Point) :
  Geo.Parallel A B C D →
  Geo.Parallel C D A B

axiom ParallelFlipRight (A B C D : Geo.Point) :
  Geo.Parallel A B C D →
  Geo.Parallel A B D C

axiom IntersectionOnSameLine (A G P B C D : Geo.Point) :
  Geo.Collinear A G P →
  IsIntersection Geo A P B C D →
  IsIntersection Geo P G B C D

axiom CollinearTrans (A G P D : Geo.Point) :
  Geo.Collinear A G P →
  Geo.Collinear P D G →
  Geo.Collinear A G D

theorem ParallelogramOfParallel (Geo : Geometry.Geo) (A B C D : Geo.Point) :
    Geo.Parallel A B C D →
    Geo.Parallel A D B C →
    IsParallelogram Geo A B C D := by
  intro h1 h2
  exact And.intro h1 h2

theorem FinlayStep1 (A B C E F G P : Geo.Point)
    (hE : IsMidpoint Geo E A C)
    (hF : IsMidpoint Geo F A B)
    (hG : IsMidpoint Geo G A P)
    (hCFG : Geo.Collinear C F G)
    (hBEG : Geo.Collinear B E G) :
    Geo.Parallel C G B P ∧ Geo.Parallel B G C P := by
  constructor
  · have hFG : Geo.Parallel F G B P := MidsegmentTheorem Geo A B P F G hF hG
    exact ParallelCollinearLeft Geo F G C B P hFG hCFG
  · have hEG : Geo.Parallel E G C P := MidsegmentTheorem Geo A C P E G hE hG
    exact ParallelCollinearLeft Geo E G B C P hEG hBEG

theorem FinlayStep2 (B C G P : Geo.Point)
    (hCG : Geo.Parallel C G B P)
    (hBG : Geo.Parallel B G C P) :
    IsParallelogram Geo B P C G := by
  have h1 : Geo.Parallel B P C G := ParallelSymm Geo C G B P hCG
  have h2 : Geo.Parallel B G P C := ParallelFlipRight Geo B G C P hBG
  exact ParallelogramOfParallel Geo B P C G h1 h2

theorem FinlayStep3 (A B C G P D : Geo.Point)
    (hAGP : Geo.Collinear A G P)
    (hInt : IsIntersection Geo A P B C D) :
    IsIntersection Geo P G B C D := by
  exact IntersectionOnSameLine Geo A G P B C D hAGP hInt

theorem FinlayStep4 (B C G P D : Geo.Point)
    (hPar : IsParallelogram Geo B P C G)
    (hInt : IsIntersection Geo P G B C D) :
    IsMidpoint Geo D B C := by
  have hPG : Geo.Collinear P D G := hInt.left
  have hBC : Geo.Collinear B D C := hInt.right
  exact ParallelogramDiagonals Geo B P C G D hPar hBC hPG

theorem FinlayStep5 (A B C D : Geo.Point) (hMid : IsMidpoint Geo D B C) :
    IsMedian Geo A D B C := by
  exact MidpointMedian Geo A B C D hMid

theorem Finlay (A B C E F G P D : Geo.Point)
    (hE : IsMidpoint Geo E A C)
    (hF : IsMidpoint Geo F A B)
    (hG : IsMidpoint Geo G A P)
    (hBE : Geo.Collinear B G E)
    (hCF : Geo.Collinear C G F)
    (hAP : Geo.Collinear A D P)
    (hBC : Geo.Collinear B D C) :
    IsMedian Geo A D B C ∧ Geo.Collinear A G D := by
  have hCFG : Geo.Collinear C F G := collinear_swap Geo C G F hCF
  have hBEG : Geo.Collinear B E G := collinear_swap Geo B G E hBE

  have hStep1 := FinlayStep1 Geo A B C E F G P hE hF hG hCFG hBEG

  have hPar : IsParallelogram Geo B P C G :=
    FinlayStep2 Geo B C G P hStep1.left hStep1.right

  have hAGP : Geo.Collinear A G P := midpoint_collinear Geo A P G hG
  have hIntAP : IsIntersection Geo A P B C D := And.intro hAP hBC

  have hIntPG : IsIntersection Geo P G B C D :=
    FinlayStep3 Geo A B C G P D hAGP hIntAP

  have hMid : IsMidpoint Geo D B C :=
    FinlayStep4 Geo B C G P D hPar hIntPG

  have hMedian : IsMedian Geo A D B C :=
    FinlayStep5 Geo A B C D hMid

  have hPG : Geo.Collinear P D G := hIntPG.left
  have hAGD : Geo.Collinear A G D := CollinearTrans Geo A G P D hAGP hPG

  exact And.intro hMedian hAGD

end Geometry
