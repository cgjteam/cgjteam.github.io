namespace Geometry

structure Geometry where
  Point : Type
  Collinear : Point → Point → Point → Prop
  Congruent : Point → Point → Point → Point → Prop
  Parallel : Point → Point → Point → Point → Prop
  AngleCongruent : Point → Point → Point → Point → Point → Point → Prop

variable {Geo : Geometry}

/-! ### Basic Structures -/

structure TriangleCongruenceResult (A B C D E F : Geo.Point) where
  sideAB : Geo.Congruent A B D E
  sideBC : Geo.Congruent B C E F
  sideAC : Geo.Congruent A C D F
  angleA : Geo.AngleCongruent B A C E D F
  angleB : Geo.AngleCongruent A B C D E F
  angleC : Geo.AngleCongruent A C B D F E

def IsMidpoint (M A B : Geo.Point) : Prop :=
  Geo.Collinear A M B ∧ Geo.Congruent A M M B

def IsOnMedian (V A B G : Geo.Point) : Prop :=
  ∃ M : Geo.Point, IsMidpoint M A B ∧ Geo.Collinear V M G

def IsCommonPointOfThreeMedians (A B C G : Geo.Point) : Prop :=
  IsOnMedian A B C G ∧ IsOnMedian B A C G ∧ IsOnMedian C A B G

def IsMedianIntersection (B E C F G : Geo.Point) : Prop :=
  Geo.Collinear B G E ∧ Geo.Collinear C G F

def IsSegmentIntersection (A P B C D : Geo.Point) : Prop :=
  Geo.Collinear A D P ∧ Geo.Collinear B D C

/-! ### Required Axioms -/

axiom CollinearRotate (A B C : Geo.Point) : Geo.Collinear A B C → Geo.Collinear A C B
axiom MedianIntersectionCollinearBGE (B E C F G : Geo.Point) : IsMedianIntersection B E C F G → Geo.Collinear B G E
axiom MedianIntersectionCollinearCGF (B E C F G : Geo.Point) : IsMedianIntersection B E C F G → Geo.Collinear C G F

axiom ThirdMedian (A B C G P D : Geo.Point) : 
  IsMidpoint G A P → IsSegmentIntersection A P B C D → IsMidpoint D B C → IsOnMedian A B C G

/-! ### Proofs -/

theorem FinlayProof (A B C E F G P D : Geo.Point)
  (hE : IsMidpoint E A C)
  (hF : IsMidpoint F A B)
  (hG : IsMedianIntersection B E C F G)
  (hSymmetric : IsMidpoint G A P)
  (hIntersection : IsSegmentIntersection A P B C D)
  (hMidBC : IsMidpoint D B C) :
  IsCommonPointOfThreeMedians A B C G := by

  -- Median from A.
  have hMedianA : IsOnMedian A B C G := by
    exact
      ThirdMedian
        A B C G P D
        hSymmetric
        hIntersection
        hMidBC

  -- Median from B.
  have hMedianB : IsOnMedian B A C G := by
    exact
      Exists.intro E (
        And.intro
          hE
          (CollinearRotate
            B G E
            (MedianIntersectionCollinearBGE
              B E C F G hG))
      )

  -- Median from C.
  have hMedianC : IsOnMedian C A B G := by
    exact
      Exists.intro F (
        And.intro
          hF
          (CollinearRotate
            C G F
            (MedianIntersectionCollinearCGF
              B E C F G hG))
      )

  exact
    And.intro
      hMedianA
      (And.intro
        hMedianB
        hMedianC)

end Geometry
